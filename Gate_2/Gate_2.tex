\section*{QUESTION}
\textbf{GATE 2010 Q.14}
\\
Consider the Z-transform $X(z) = 5z^2 + 4z^{-1} + 3,$

$ 0 < |z| < \infty$. The inverse Z-transform x[n] is

\begin{enumerate}[label={\Alph*)}]
\item $5\delta[n+2] + 3\delta[n] + 4\delta[n-1]$
\item $5\delta[n-2] + 3\delta[n] + 4\delta[n+1]$
\item $5u[n+2] + 3u[n] + 4u[n-1]$
\item $5u[n-2] + 3u[n] + 4u[n+1]$
\end{enumerate}

\section*{SOLUTION}
Given, Z-transform 
\begin{align}
    X(z) = 5z^2 + 4z^{-1} + 3 \label{eq 0}
\end{align}
ROC = $0<|z|<\infty$

If X(z) is the Z-transform of x(n), 
\begin{align}
    X(z) = \sum_{n = -\infty}^{\infty} x(n)z^{-n} \label{eq 1}
\end{align}

And if $\delta$ is the impulse response, then
\begin{align}
    x(n) = \sum_{k = -\infty}^{\infty} x(k)\delta(n-k) \label{eq 2}
\end{align}

Therefore, from \eqref{eq 1} and \eqref{eq 2}
\begin{align}
    \delta(n+a) \xleftrightarrow{z} z^{a}
\end{align}

The inverse transform
\begin{align}
    x(n) &= Z^{-1}[X(z)]\\
    &= 5\delta[n+2] + 3\delta[n] + 4\delta[n-1]
\end{align}

Option (A) is correct.
